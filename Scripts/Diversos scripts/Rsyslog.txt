#logrotate
vi  /etc/logrotate.d/samba
/var/log/samba/log.[b-z]* {
    notifempty
    olddir /var/log/samba/old
    missingok
    sharedscripts
    copytruncate
}

/var/log/samba/audit.log {
   weekly
   missingok
   rotate 90
   postrotate
      /bin/kill -HUP `cat /var/run/syslogd.pid 2> /dev/null` 2> /dev/null || true
   endscript
   compress
   notifempty
}
