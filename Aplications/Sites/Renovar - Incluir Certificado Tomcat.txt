1- Converter o certificado
openssl pkcs12 -export -in fullchain.pem -inkey privkey.pem -out pkcs.p12 -name tomcat -passout pass:ctmatic

2-Importar o certificado para uma keystore
./keytool -importkeystore -deststorepass ctmatic -destkeypass ctmatic -destkeystore .keystore -srckeystore pkcs.p12 -srcstoretype PKCS12 -srcstorepass ctmatic -alias tomcat

3-Listar as keystore para validar
./keytool -list -v -keystore .keystore -storepass ctmatic

4-Backup da keystore atual identificada no caminho do server.xml do Tomcat
cp /opt/contador-online/apache-tomcat-7.0.82/webapps/.keystore /root/

5-Copiar a nova keystore para o path identificado no path do server.xml do Tomcat
cp .keystore /opt/contador-online/apache-tomcat-7.0.82/webapps/
